\pronounce{$\Gamma \mapsto \Omega$}{$\Gamma$ is satisfied by $\Omega$}
\begin{mathpar}

\inferrule*[right=$\mapsto$ID]{ }{\cdot \mapsto \cdot}

\inferrule*[right=$\mapsto$Var]{ \Gamma \mapsto \Omega \\ [\Omega]A = [\Omega]A' }{\Gamma,x : A \mapsto \Omega,x:A'}

\inferrule*[right=$\mapsto$Uvar]{\Gamma\mapsto\Omega}{\Gamma,\alpha \mapsto \Omega,\alpha}

\inferrule*[right=$\mapsto$Solve]{\Gamma\mapsto\Omega \\ \Gamma \vdash \sigma_l \sigbndl \tau \dashv \Theta \\ \Gamma \vdash \tau \sigbndr \sigma_u \dashv \Theta' \\ \Theta \mapsto \Omega \\ \Theta' \mapsto\Omega}{\Gamma,\bound{\sigma_l}{\alphahat}{\sigma_u} \mapsto \Omega,\alphahat=\tau}

\inferrule*[right=$\mapsto$Add]{\Gamma \mapsto \Omega}{\Gamma\mapsto\Omega,\alphahat=\tau}

\inferrule*[right=$\mapsto$AddSolved]{\Gamma\mapsto\Omega}{\Gamma\mapsto\Omega,\alphahat=\tau}

\inferrule*[right=$\mapsto$Marker]{\Gamma\mapsto\Omega}{\Gamma,\marker{\alphahat} \mapsto \Omega,\marker{\alphahat}}
\end{mathpar}
\pronounce{$\Gamma \hookrightarrow \Delta$}{$\Delta$ extends $\Omega$}
\begin{mathpar}

\inferrule*[right=$\hookrightarrow$ID]{ }{\cdot \hookrightarrow \cdot}

\inferrule*[right=$\hookrightarrow$Var]{ \Gamma \hookrightarrow \Delta}{\Gamma,x : A \hookrightarrow \Delta,x:A}

\inferrule*[right=$\hookrightarrow$Uvar]{\Gamma\hookrightarrow\Omega}{\Gamma,\alpha \hookrightarrow \Omega,\alpha}

\inferrule*[right=$\hookrightarrow$Constrain]{
\Gamma\hookrightarrow\Delta \\ 
\Gamma \vdash \sigma_l \glb \tau_l = \sigma_l \dashv \Theta_1 \\ 
\Gamma \vdash \sigma_u \lub \tau_u = \sigma_u \dashv \Theta_2 \\
\rho'_l \subseteq \rho_l\\ 
\rho'_u \subseteq \rho_u\\
}{
	\Gamma,\bound{\pair{\rho'_l}{\tau'_l}}{\alphahat}{\pair{\rho'_u}{\tau'_u}} 
		\hookrightarrow 
	\Delta,\bound{\pair{\rho_l}{\tau_l}}{\alphahat}{\pair{\rho_u}{\tau_u}}
}

\inferrule*[right=$\hookrightarrow$Add]{\Gamma \hookrightarrow \Delta}{
\Gamma\hookrightarrow\Delta,\bound{\sigma_l}{\alphahat}{\sigma_t}}

\inferrule*[right=$\hookrightarrow$Marker]{\Gamma\hookrightarrow\Omega}{\Gamma,\marker{\alphahat} \hookrightarrow \Omega,\marker{\alphahat}}
\end{mathpar}
