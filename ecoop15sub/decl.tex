\inferrule*[right=$\leq$Var]{ \alpha \in \Psi }{ \Psi \vdash \alpha \leq \alpha}

\inferrule*[right=$\leq$Unit]{ }{ \Psi \vdash 1 \leq 1}

\inferrule*[right=$\leq\rightarrow$]{\Psi \vdash B_1 \leq A_1 \\ \Psi \vdash A_2 \leq A_2}{\Psi \vdash A_1 \rightarrow A_2 \leq B_1 \rightarrow B_2}

\inferrule*[right=$\leq\forall$L]{\Psi \vdash \tau \\ \Psi \vdash [\tau/\alpha]A \leq B}{\Psi \vdash \forall\alpha.A \leq B}\\

\inferrule*[right=$\leq\forall$R]{\Psi,\beta \vdash A \leq B}{\Psi \vdash A \leq \forall\beta.B}\\

\inferrule*[right=$\leq$Gen]{ X < Y }{\Psi \vdash X \leq Y}\\
