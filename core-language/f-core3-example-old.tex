\begin{lstlisting}
type t
	meth add : int -> t
	meth get : int
	
class c
	var f : int
	meth add : int -> c =
		fn x : int =>
			this.f = this.f + x
			this
	meth get : int
	meth equals : c -> bool =
		fn other : c =>
			this.f == other.get // cannot access other.f in type c
								// because this core doesn't support ADTs
	class var nc : int = 1
	class meth make : int -> c =
		fn x:int =>
			nc = nc + 1
			new
				var f:int = x
				meth get : int = this.f
			
val o : c = c.make(4)
val o2 : t = o.add(2)
var x6 : int = o.get
\end{lstlisting}
