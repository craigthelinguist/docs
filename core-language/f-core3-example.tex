class Option
	var quantity : int = 0
	var price : int = 0
	meth exercise : int =
		this.quantity * this.price

	class var totalQuantityIssued : int = 0
	class meth issue : int -> int -> Option =
		fn q : int =>
			fn p : int =>
				totalQuantityIssued = 
					totalQuantityIssued + q
				new
					var quantity : int = q
					var price : int = p

var optn : Option = Option.issue(100, 50)
var ret : int = optn.exercise
