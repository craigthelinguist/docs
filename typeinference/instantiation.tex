\pronounce{$\Gamma \vdash \alphahat \instl \tau \dashv \Delta$}{Against $\Gamma$, let $\alphahat$ be a subtype of $\tau$ in $\Delta$}
\pronounce{$\Gamma \vdash \tau \instl \alphahat \dashv \Delta$}{Against $\Gamma$, let $\alphahat$ be a supertype of $\tau$ in $\Delta$}
\begin{mathpar}
\inferrule*[right=InstLAbs]{ 
	\Gamma \vdash \tau \\
	\Gamma \vdash \botbound \st \tau \dashv \Delta
}
{ 
		\Gamma, 
		\bound{\botbound}{\alphahat}{\topbound}, 
		\Gamma' 
	\vdash 
		\alphahat \instl \tau 
	\dashv 
		\Delta, 
		\bound{\botbound}{\alphahat}{\topbound \vee \tau}, 
		\Gamma'
}

\inferrule*[right=InstLReach]{
	\Gamma \vdash \alphahat \st \topbound \dashv \Delta
}
{
	\Gamma[\alphahat],
		\bound{\botbound}{\hat{\beta}}{\topbound},
	\Gamma'
	\vdash 
		\alphahat \instl \hat{\beta} 
	\dashv 
	\Delta,
		\bound{\botbound \wedge \alphahat}{\hat{\beta}}{\topbound},\Gamma'
}

\inferrule*[right=InstLArrow]{
	\Gamma_u,\hat{\alpha}_2, \hat{\alpha}_1 \vdash \botbound \st (\hat{\alpha}_1 \rightarrow \hat{\alpha}_2)\dashv \Theta_1\\
		\Theta_1
	\vdash
		A_1 \instr \alphahat_1
	\dashv
		\Theta_2
		\\
		\Theta_2
	\vdash
		\alphahat_2 \instl A_2
	\dashv
		\Delta
}
{	
	\Gamma_u,\bound{\botbound}{\alphahat}{\topbound},\Gamma_l \vdash \alphahat \instl A_1 \rightarrow A_2 \dashv \Delta,\bound{\botbound}{\alphahat}{\topbound\vee (\alphahat_1 \rightarrow \alphahat_2)},\Gamma_l
}

\inferrule*[right=InstLAIIL]{\Gamma[\alphahat],\betahat \vdash \alphahat \instl [\alphahat/\alpha]B \dashv \Delta,\betahat,\Delta'}
{
	\Gamma[\alphahat] \vdash \alphahat \instl \forall\beta.B \dashv \Delta
}

\inferrule*[right=InstRAbs]{ \Gamma \vdash \tau \\ 
	\Gamma \vdash \tau \st \topbound \dashv \Delta}
{ 
\Gamma, \bound{\botbound}{\alphahat}{\topbound}, \Gamma' \vdash \tau \instr \alphahat \dashv \Delta, \bound{\botbound \wedge \tau}{\alphahat}{\topbound}, \Gamma'
}

\inferrule*[right=InstRReach]{
	\Gamma \vdash \botbound \st \alphahat \dashv \Delta
}
{
	\Gamma[\alphahat],
		\bound{\botbound}{\hat{\beta}}{\topbound},
	\Gamma'
	\vdash 
		\hat{\beta} \instr \alphahat
	\dashv 
	\Delta,
		\bound{\botbound}{\hat{\beta}}{\topbound \vee \alphahat},\Gamma'
}

\inferrule*[right=InstRArrow]{
	\Gamma,\hat{\alpha}_2, \hat{\alpha}_1,\bound{\botbound}{\alphahat}{\topbound}] \vdash (\hat{\alpha}_1 \rightarrow \hat{\alpha}_2) \st \topbound \dashv \Theta_1\\
		\Theta_1
	\vdash
		\alphahat_1 \instl A_1
	\dashv
		\Theta_2
		\\
		\Theta_2
	\vdash
		A_2 \instr \alphahat_2
	\dashv
		\Delta
}
{	
	\Gamma,\bound{\botbound}{\alphahat}{\topbound},\Gamma' \vdash A_1 \rightarrow A_2 \instr \alphahat \dashv \Delta,\botbound \wedge (\alphahat_1 \rightarrow \alphahat_2),\Gamma'
}

\inferrule*[right=InstRAIIL]{\Gamma[\alphahat],\marker{\betahat},\betahat \vdash [\betahat/\beta]B \instr \alphahat \dashv \Delta,\marker{\betahat},\Delta'}
{
	\Gamma[\alphahat] \vdash \forall\beta.B \instr \alphahat \dashv \Delta
}
\end{mathpar}
