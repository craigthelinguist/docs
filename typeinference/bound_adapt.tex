\pronounce{$\tau \barwedge \tau' = \tau''$}{The least upper bound of $\tau$ and $\tau'$ is $\tau''$}
\begin{mathpar}
\inferrule*[right=$\barwedge\rightarrow$]{ 
	\Gamma \vdash \tau_1 \veebar \tau_3 = \tau' \dashv \Theta\\
	\Theta \vdash \tau_2 \barwedge \tau_4 = \tau'' \dashv \Delta
	}{\Gamma \vdash \tau_1 \rightarrow \tau_2 \barwedge \tau_3 \rightarrow \tau_4 = \tau' \rightarrow \tau'' \dashv \Delta}

\inferrule*[right=$\barwedge L$]{ \Gamma \vdash \tau_1 \st \tau_2 \dashv \Delta}{\Gamma \vdash \tau_1 \barwedge \tau_2 = \tau_2 \dashv \Delta}

\inferrule*[right=$\barwedge R$]{ \Gamma \vdash \tau_2 \st \tau_1 \dashv \Delta}{\Gamma \vdash \tau_1 \barwedge \tau_2 = \tau_1 \dashv \Delta}


\inferrule*[right=$\barwedge \exists$]{ 
	\Gamma,\alphahat',\betahat \vdash \alphahat \instr \betahat \dashv \Theta\\
	\Theta \vdash \alphahat' \instr \betahat \dashv \Delta\\
}{
	\Gamma[\alphahat],\alphahat',\Gamma' \vdash \alphahat \barwedge \alphahat' = \pair{\rho}{\betahat} \dashv \Delta,\Gamma'
}
\end{mathpar}
\pronounce{$\tau \veebar \tau' = \tau''$}{The greatest lower bound of $\tau$ and $\tau'$ is $\tau''$}
\begin{mathpar}
\inferrule*[right=$\veebar\rightarrow$]{ 
	\Gamma \vdash \tau_1 \barwedge \tau_3 = \tau' \dashv \Theta\\
	\Theta \vdash \tau_2 \veebar \tau_4 = \tau'' \dashv \Delta
	}{\Gamma \vdash \tau_1 \rightarrow \tau_2 \veebar \tau_3 \rightarrow \tau_4 = \tau' \rightarrow \tau'' \dashv \Delta}

\inferrule*[right=$\veebar L$]{ \Gamma \vdash \tau_1 \st \tau_2 \dashv \Delta}{\Gamma \vdash \tau_1 \veebar \tau_2 = \tau_1 \dashv \Delta}

\inferrule*[right=$\veebar R$]{ \Gamma \vdash \tau_2 \st \tau_1 \dashv \Delta}{\Gamma \vdash \tau_1 \veebar \tau_2 = \tau_2 \dashv \Delta}


\inferrule*[right=$\veebar \exists$]{ 
	\Gamma,\alphahat',\betahat \vdash \betahat \instr \alphahat \dashv \Theta\\
	\Theta \vdash \betahat \instr \alphahat' \dashv \Delta\\
}{
	\Gamma[\alphahat],\alphahat',\Gamma' \vdash \alphahat \veebar \alphahat' = \pair{\rho}{\betahat} \dashv \Delta,\Gamma'
}

\end{mathpar}

\begin{mathpar}

\end{mathpar}
